-----------------------main---------------------------
import java.util.Random;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
    }

    public void onClickButton(View v) {
        //obteniendo un numero aleatorio entre 0 y 4
        int color = (int) (new Random().nextDouble() * 5);
        //obteniendo el FragmentTransaction
        FragmentTransaction transaction = getSupportFragmentManager().beginTransaction();
        //creando el fragmento
        ColorFragment fragment = ColorFragment.getInstance(color);
        //reemplazando el fragmento
        transaction.replace(R.id.managerFragmentLayout, fragment);
        //realizando el cambio
        transaction.commit();
    }

}

----------------------------color fragment---------------------
public class ColorFragment extends Fragment {

    LinearLayout layout;

    TextView textColor;

    public static ColorFragment getInstance(int color) {
        ColorFragment fragment = new ColorFragment();

        Bundle params = new Bundle();
        params.putInt("color", color);

        fragment.setArguments(params);

        return fragment;
    }


    private void changeColor() {
        String[] colors = getResources().getStringArray(R.array.colors);

        int color = 0;

        if (getArguments() == null)
            return;
        else
            color = getArguments().getInt("color");

        Log.i("InfoColor", String.valueOf(color));

        textColor.setText(colors[color]);

        switch (color) {
            case 0:
                textColor.setTextColor(getResources().getColor(R.color.darkYellow));
                layout.setBackgroundColor(getResources().getColor(R.color.ligthYellow));
                break;
            case 1:
                textColor.setTextColor(getResources().getColor(R.color.darkBlue));
                layout.setBackgroundColor(getResources().getColor(R.color.ligthBlue));
                break;
            case 2:
                textColor.setTextColor(getResources().getColor(R.color.darkRed));
                layout.setBackgroundColor(getResources().getColor(R.color.ligthRed));
                break;
            case 3:
                textColor.setTextColor(getResources().getColor(R.color.darkGreen));
                layout.setBackgroundColor(getResources().getColor(R.color.ligthGreen));
                break;
            case 4:
                textColor.setTextColor(getResources().getColor(R.color.darkFuchsia));
                layout.setBackgroundColor(getResources().getColor(R.color.ligthFuchsia));
                break;
        }
    }

    @Override
    public View onCreateView(LayoutInflater inflater, ViewGroup container,
                             Bundle savedInstanceState) {
        return inflater.inflate(R.layout.fragment_color, container, false);
    }

    @Override
    public void onActivityCreated(Bundle savedInstanceState) {
        super.onActivityCreated(savedInstanceState);

        layout = (LinearLayout) getView().findViewById(R.id.backLayout);
        textColor = (TextView) getView().findViewById(R.id.textColor);

        changeColor();
    }
}
