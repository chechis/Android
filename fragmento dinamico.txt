----------------------------main----------

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        FragmentManager fragmentManager = getFragmentManager();
        FragmentTransaction fragmentTransaction = fragmentManager.beginTransaction();

        Categorias categorias = new Categorias();
        fragmentTransaction.add(R.id.contenedor1, categorias);
        fragmentTransaction.commit();
    }

    public void mostrarEmails(View view){
        FragmentManager fragmentManager = getFragmentManager();
        FragmentTransaction fragmentTransaction = fragmentManager.beginTransaction();

        Email email = new Email();
        fragmentTransaction.add(R.id.contenedor2, email);
        fragmentTransaction.commit();
    }
}

--------------------------Email----------------------------

public class Email extends Fragment{

    @Override
    public View onCreateView(LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {

        View view = inflater.inflate(R.layout.email, container, false);

        String [] email = new String[]{"Darth Vader", "Ezio Auditore", "Leonardo Da Vinci", "Papa Francisco", "Larry Page"};

        ArrayAdapter arrayAdapter = new ArrayAdapter(getActivity(), android.R.layout.simple_list_item_1, email);
        ListView listView = (ListView) view.findViewById(R.id.email);
        listView.setAdapter(arrayAdapter);

        return view;
    }
}

public class Categorias extends Fragment {

    @Override

        ListView listView = (ListView) view.findViewById(R.id.lista_categorias);

        return view;
    }
}
