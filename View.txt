-----------------------------main-----------------------------
    private RadioGroup rgOpciones;
    private TextView txtRadioS;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        rgOpciones = (RadioGroup) findViewById(R.id.rg_opciones);
        txtRadioS = (TextView) findViewById(R.id.txt_radio_s);

        rgOpciones.setOnCheckedChangeListener(new RadioGroup.OnCheckedChangeListener() {
            @Override
            public void onCheckedChanged(RadioGroup group, int checkedId) {

                switch (checkedId){
                    case R.id.rb_opcion1:
                        txtRadioS.setText(getResources().getString(R.string.opcion1)+" seleccionado");
                        break;
                    case R.id.rb_opcion2:
                        txtRadioS.setText(getResources().getString(R.string.opcion2)+" seleccionado");
                        break;
                    case R.id.rb_opcion3:
                        txtRadioS.setText(getResources().getString(R.string.opcion3)+" seleccionado");
                        break;
                }
            }
        });
    }
}
