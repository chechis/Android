public class MainActivity extends AppCompatActivity {

    private int actualPage = 1;

    private static final int MAX_NUMBER_PAGES = 3;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        cambiarPagina(actualPage);
    }

    private void cambiarPagina(int page) {
        Fragment fragment = getSupportFragmentManager().findFragmentById(R.id.libroFragment);

        if (fragment != null && fragment.getView() != null) {
            TextView textPagina = (TextView) fragment.getView().findViewById(R.id.textPagina);
            TextView textNumberoPagina= (TextView) fragment.getView().findViewById(R.id.textNumberoDePagina);

            String formatoNumeroPagina = String.format("Página %d", page);
            String[] paginas = getResources().getStringArray(R.array.paginas);

            textNumberoPagina.setText(formatoNumeroPagina);
            textPagina.setText(paginas[page - 1]);
        } else {
            Log.i(MainActivity.class.getSimpleName(), "el fragmento o la vista es nula");
        }

        Log.i(MainActivity.class.getSimpleName(), "entro en el metodo.");
    }

    public void onClickAdelanteButton(View view) {
        if (actualPage < MAX_NUMBER_PAGES) {
            cambiarPagina(++actualPage);
            cambiarEstadoDelEditText();
        } else {
            Snackbar.make(view, "No hay más páginas.", Snackbar.LENGTH_LONG).show();
        }
    }

    private void cambiarEstadoDelEditText() {
        EditText editText = (EditText) findViewById(R.id.editBuscar);
        if (editText != null)
            editText.setText(String.valueOf(actualPage));
    }

    public void onClickAtrasButton(View view) {
        if (actualPage > 1) {
            cambiarPagina(--actualPage);
            cambiarEstadoDelEditText();
        } else {
            Snackbar.make(view, "No puedes retroceder más", Snackbar.LENGTH_LONG).show();
        }
    }

    public void onClickIrButton(View view) {
        EditText editText = (EditText) findViewById(R.id.editBuscar);
        int page = 1;
        if (editText != null)
            page = Integer.parseInt(editText.getText().toString());

        if (page > 1 && page < 3) {
            actualPage = page;
            cambiarPagina(actualPage);
        } else {
            Snackbar.make(view, "No existe esa página.", Snackbar.LENGTH_LONG).show();
        }
    }

}

public class BookFragment extends Fragment {

    public static BookFragment getInstance(int page) {
        BookFragment bookFragment = new BookFragment();

        Bundle params = new Bundle();
        params.putInt("page", page);

        bookFragment.setArguments(params);
        return bookFragment;
    }

    @Nullable
    @Override
    public View onCreateView(LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {
        return inflater.inflate(R.layout.fragment_book, container);
    }

}
