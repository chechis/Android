--------------------------------Main--------------------

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
    }

    public void onClickEnviarMensaje(View view) {
        EditText editMensaje = (EditText) findViewById(R.id.edit_mensaje);
        String mensaje = editMensaje != null ? editMensaje.getText().toString() : "No hay mensaje";

        Fragment fragmento = getSupportFragmentManager().findFragmentByTag("tagFragmentoMensaje");
        if (fragmento instanceof MensajeFragment) {
            MensajeFragment mensajeFragmento = (MensajeFragment) fragmento;
            mensajeFragmento.getTxtMensaje().setText(mensaje);
        }
    }

}

---------------------------------Mensaje fragment---------------
public class MensajeFragment extends Fragment {

    private TextView txtMensaje;

    @Nullable
    @Override
    public View onCreateView(LayoutInflater inflater, @Nullable ViewGroup container,
                             @Nullable Bundle savedInstanceState) {
        return inflater.inflate(R.layout.fragment_mensaje, container, false);
    }

    @Override
    public void onViewCreated(View view, @Nullable Bundle savedInstanceState) {
        super.onViewCreated(view, savedInstanceState);

        this.txtMensaje = (TextView) view.findViewById(R.id.txt_mensaje);
    }

    public TextView getTxtMensaje() {
        return txtMensaje;
    }
}
